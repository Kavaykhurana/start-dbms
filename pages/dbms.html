Normalization Walkthrough

Use this visual path when the teacher asks how the schema became clean and non-redundant.

UNF → BCNF
UNF One messy investment sheet had startup, sector, investor, funding, metrics, market, partner, and risk columns repeated together.
1NF Repeated groups were made atomic: one row per startup, one row per investment, one row per monthly metric.
2NF Partial dependencies were removed: investor details moved to investors, sector details moved to sectors.
3NF Transitive dependencies were removed: market snapshots, risk scores, and partnerships became separate child tables.
BCNF Every determinant is a candidate key: lookup tables hold master data and transaction tables depend only on their keys.

View 1: vw_TopStartups 
CREATE OR REPLACE VIEW vw_TopStartups AS
SELECT s.startup_id, s.startup_name, sec.sector_name, s.funding_stage,
       ROUND(fm.ltv_inr / NULLIF(fm.cac_inr, 0), 2) AS ltv_cac_ratio,
       fm.runway_months, fm.churn_rate
FROM startups s
JOIN sectors sec ON sec.sector_id = s.sector_id
JOIN financial_metrics fm ON fm.startup_id = s.startup_id
WHERE fm.metric_month = (
    SELECT MAX(metric_month)
    FROM financial_metrics
    WHERE startup_id = s.startup_id
)
AND (fm.ltv_inr / NULLIF(fm.cac_inr, 0)) >= 4
AND fm.runway_months >= 9;
View 2: vw_HighRiskAlerts 
CREATE OR REPLACE VIEW vw_HighRiskAlerts AS
SELECT s.startup_name, sec.sector_name, fm.burn_rate_inr,
       fm.runway_months, fm.churn_rate, AVG(ra.risk_score) AS avg_risk_score
FROM startups s
JOIN sectors sec ON sec.sector_id = s.sector_id
JOIN financial_metrics fm ON fm.startup_id = s.startup_id
LEFT JOIN investments i ON i.startup_id = s.startup_id
LEFT JOIN risk_analysis ra ON ra.investment_id = i.investment_id
WHERE fm.runway_months < 6
GROUP BY s.startup_name, sec.sector_name, fm.burn_rate_inr, fm.runway_months, fm.churn_rate;
Trigger 1: trig_EquityCheck 
CREATE TRIGGER trig_EquityCheck
BEFORE INSERT ON investments
FOR EACH ROW
BEGIN
    DECLARE v_total_equity DECIMAL(7,2);

    SELECT COALESCE(SUM(equity_percentage), 0)
    INTO v_total_equity
    FROM investments
    WHERE startup_id = NEW.startup_id
      AND status = 'Active';

    IF v_total_equity + NEW.equity_percentage > 100 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Equity cannot exceed 100 percent.';
    END IF;
END;
Trigger 2: trig_RiskUpdate 
CREATE TRIGGER trig_RiskUpdate
AFTER UPDATE ON financial_metrics
FOR EACH ROW
BEGIN
    IF NEW.burn_rate_inr <> OLD.burn_rate_inr
       OR NEW.churn_rate <> OLD.churn_rate
       OR NEW.runway_months <> OLD.runway_months THEN
        UPDATE risk_analysis ra
        JOIN investments i ON i.investment_id = ra.investment_id
        SET ra.risk_score = LEAST(100,
            CASE WHEN NEW.runway_months < 6 THEN 35 ELSE 10 END
            + LEAST(30, NEW.churn_rate * 4)
        )
        WHERE i.startup_id = NEW.startup_id;
    END IF;
END;
Procedure 1: sp_CalculateROI 
CREATE PROCEDURE sp_CalculateROI (
    IN p_investment_id BIGINT,
    OUT p_roi_percentage DECIMAL(10,2)
)
BEGIN
    SELECT ROUND(
        ((fm.net_profit_inr * 12 * 5 * (i.equity_percentage / 100))
        / NULLIF(i.invested_amount_inr, 0)) * 100,
        2
    )
    INTO p_roi_percentage
    FROM investments i
    JOIN financial_metrics fm ON fm.startup_id = i.startup_id
    WHERE i.investment_id = p_investment_id
    ORDER BY fm.metric_month DESC
    LIMIT 1;
END;
Procedure 2: sp_GetRecommendation 
CREATE PROCEDURE sp_GetRecommendation()
BEGIN
    SELECT startup_name, sector_name, funding_stage,
           ROUND(ltv_cac_ratio, 2) AS ltv_cac_ratio,
           ROUND(recommendation_score / 10, 2) AS decision_score
    FROM (
        SELECT s.startup_name, sec.sector_name, s.funding_stage,
               fm.ltv_inr / NULLIF(fm.cac_inr, 0) AS ltv_cac_ratio,
               0.50 * LEAST(100, (fm.ltv_inr / NULLIF(fm.cac_inr, 0)) * 16.67)
               + 0.30 * (100 - COALESCE(rs.latest_risk_score, 50)) AS recommendation_score
        FROM startups s
        JOIN sectors sec ON sec.sector_id = s.sector_id
        JOIN financial_metrics fm ON fm.startup_id = s.startup_id
        LEFT JOIN (
            SELECT i.startup_id, AVG(ra.risk_score) AS latest_risk_score
            FROM investments i
            JOIN risk_analysis ra ON ra.investment_id = i.investment_id
            GROUP BY i.startup_id
        ) rs ON rs.startup_id = s.startup_id
        WHERE fm.metric_month = (
            SELECT MAX(metric_month)
            FROM financial_metrics
            WHERE startup_id = s.startup_id
        )
    ) ranked
    ORDER BY recommendation_score DESC
    LIMIT 5;
END;
Complex Join: highest ROI Fintech investors 
SELECT inv.investor_name, s.startup_name, sec.sector_name,
       i.funding_round, fm.net_profit_inr, ra.risk_score,
       ROUND(((fm.net_profit_inr * 12 * 5 * (i.equity_percentage / 100))
       / i.invested_amount_inr) * 100, 2) AS roi_proxy_pct
FROM startups s
JOIN sectors sec ON sec.sector_id = s.sector_id
JOIN investments i ON i.startup_id = s.startup_id
JOIN investors inv ON inv.investor_id = i.investor_id
JOIN financial_metrics fm ON fm.startup_id = s.startup_id
LEFT JOIN risk_analysis ra ON ra.investment_id = i.investment_id
WHERE sec.sector_name = 'Fintech'
ORDER BY roi_proxy_pct DESC;
Correlated Subquery: sector-average burn rate 
SELECT s.startup_name, sec.sector_name, fm.burn_rate_inr
FROM startups s
JOIN sectors sec ON sec.sector_id = s.sector_id
JOIN financial_metrics fm ON fm.startup_id = s.startup_id
WHERE fm.burn_rate_inr > (
    SELECT AVG(fm_sector.burn_rate_inr)
    FROM startups s_sector
    JOIN financial_metrics fm_sector
      ON fm_sector.startup_id = s_sector.startup_id
    WHERE s_sector.sector_id = s.sector_id
);

Trigger: Equity Check

Blocks a new investment if active equity for a startup crosses 100%.

CREATE OR REPLACE TRIGGER trig_EquityCheck
FOR INSERT OR UPDATE OF equity_percentage, startup_id, status
ON investments
COMPOUND TRIGGER
    TYPE t_startup_ids IS TABLE OF investments.startup_id%TYPE INDEX BY PLS_INTEGER;
    g_startup_ids t_startup_ids;
    g_count PLS_INTEGER := 0;

    BEFORE EACH ROW IS
    BEGIN
        g_count := g_count + 1;
        g_startup_ids(g_count) := :NEW.startup_id;
    END BEFORE EACH ROW;

    AFTER STATEMENT IS
        v_total_equity NUMBER(7,2);
    BEGIN
        FOR idx IN 1 .. g_count LOOP
            SELECT NVL(SUM(equity_percentage), 0)
            INTO v_total_equity
            FROM investments
            WHERE startup_id = g_startup_ids(idx)
              AND status = 'Active';

            IF v_total_equity > 100 THEN
                RAISE_APPLICATION_ERROR(-20001, 'Equity cannot exceed 100 percent.');
            END IF;
        END LOOP;
    END AFTER STATEMENT;
END trig_EquityCheck;

Procedure: ROI

Calculates ROI from latest net profit and investor equity.

CREATE OR REPLACE PROCEDURE sp_CalculateROI (
    p_investment_id IN NUMBER,
    p_roi_percentage OUT NUMBER
) AS
BEGIN
    SELECT ROUND(
        ((fm.net_profit_inr * 12 * 5 * (i.equity_percentage / 100))
        / NULLIF(i.invested_amount_inr, 0)) * 100,
        2
    )
    INTO p_roi_percentage
    FROM investments i
    JOIN financial_metrics fm
      ON fm.startup_id = i.startup_id
    WHERE i.investment_id = p_investment_id
      AND fm.metric_month = (
          SELECT MAX(metric_month)
          FROM financial_metrics
          WHERE startup_id = i.startup_id
      );
END;

Normalization Made Simple

1NFEvery cell stores one atomic value.
2NFMetrics depend on the complete metric row, not part of a combined key.
3NFSector, investor, startup, and metric data are separated to remove transitive dependency.
BCNFUnique determinants like startup_name and sector_name are candidate keys.
